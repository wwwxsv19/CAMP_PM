#include <stdio.h>
#include <stdlib.h>
int maxh(int* num, int n) {
	int i ;
	int maxn = -1; //범위에 음수도 포함될 때는 0번째 인덱스의 값으로 실행하게 설정해두면 도움이 된다!
		for(i=0; i<n; i++){
			if(maxn <= *(num+i))
				maxn=*(num+i);
		}
	return maxn;
}
int main(){
  int n, i;
  int* num;
	scanf("%d", &n) ;
	
	num = (int*)malloc(sizeof(int)*n);
	
	for(i=0; i<n; i++){
		scanf("%d", (num+i));
	}
	printf("%d\n", maxh(num, n)) ;
	
	free(num);
  return 0; 
}
