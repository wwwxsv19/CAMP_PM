#include <stdio.h>
int main() {
	char* s = "world";
	int i;
	printf("%08X %08X %08X %s\n", &s, s, &s[0], s); //포인터 변수 s의 주소, s 가 가리키는 주소, " , s의 값
	for(i=0; s[i]!='\0'; i++){
		printf("%08X %c\n", &s[i], s[i]); //. world 각 요소의 주소와 그 값
	}
	return 0;
	
}

#include <stdio.h>
void outputArray(int* p, int n){
	int i;
	for(i=0; i<n; i++){
		printf("%d %d\n", p[i], *(p+i)); //size만큼 (3) array 의 i번째 인덱스의 값을 배열ver 과 포인터ver 로 출력
	}
}
int main(){
	int array[3] = {10, 20, 30};
	int size = sizeof(array) / sizeof(array[0]); //array의 전체 크기 (4바이트*3=12바이트) 나누기 array0번째인덱스의 크기 4바이트, 즉 12/4=3.
	outputArray(array, size); //array의 시작 주소 와 3
}
