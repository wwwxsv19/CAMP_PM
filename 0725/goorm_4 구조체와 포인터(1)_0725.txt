#include<stdio.h>
#include<string.h>
#define NUM 3
typedef struct Person{//전역에 구조체 정의
	char name[10];
	int age;
}Ps;

void inputData( Ps *p );
void outputData( Ps *p );

int main(){
	Ps p[3];
	int i;
		
	for(i=0; i<3; i++){
		inputData(&p[i]);
	}
	
	for(i=0; i<3; i++){
		outputData(&p[i]);
	}
	return 0;
}

void inputData( Ps *p ){
	scanf("%s", p->name);
	scanf("%d", &p->age);
}
void outputData( Ps *p ){
	printf("%s, ", p->name);
	printf("%d\n", p->age);
}
