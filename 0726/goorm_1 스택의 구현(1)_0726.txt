#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#define MAX 100
typedef int element; //int를 element로 재정의

typedef struct STACK{
	int top;
	int data[MAX];
}stk;

stk s;

void initStack(){
	s.top = -1;
}

bool isEmpty(){
	return (s.top == -1);
}
bool isFull(){
	return (s.top ==(MAX-1));
}

void push(element item){
	if (isFull()) return;
	else s.data[++s.top] = item;
}

element pop(){ //반환후 제거
	if(isEmpty()) exit(1);
	else return s.data[s.top--];
}

element peek(){//반환만
	if (isEmpty()) exit(1);
	else return s.data[s.top];
}

int main(){	
	push(1);
	push(2);
	push(3);
	printf("%d\n",pop());
	printf("%d\n",peek());
	printf("%d\n",pop());
	return 0;
}
